// Remove item
function removeItem(e){
  if(e.target.classList.contains('delete')){
    if(confirm('Are You Sure?')){
      // var li = e.target.parentElement;
      var li = items.lastElementChild;
      itemList.removeChild(li);
    }
  }
}
